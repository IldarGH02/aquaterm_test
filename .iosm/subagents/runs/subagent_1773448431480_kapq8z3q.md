```typescript
// lib/validation/phone.validator.ts
export class PhoneValidator {
  private static readonly REGEX = /^(\+7|8)[\s-]?\(?\d{3}\)?[\s-]?\d{3}[\s-]?\d{2}[\s-]?\d{2}$/

  static validate(phone: string): boolean {
    const cleaned = phone.replace(/\s/g, '')
    return this.REGEX.test(cleaned)
  }

  static format(phone: string): string {
    const cleaned = phone.replace(/\D/g, '')
    if (cleaned.length !== 11) return phone
    return `+${cleaned[0]} (${cleaned.slice(1,4)}) ${cleaned.slice(4,7)}-${cleaned.slice(7,9)}-${cleaned.slice(9,11)}`
  }

  static isComplete(phone: string): boolean {
    return phone.replace(/\D/g, '').length === 11
  }
}

// Использование:
import { PhoneValidator } from '@/lib/validation/phone.validator'

PhoneValidator.validate('+7 (920) 123-45-67') // true
PhoneValidator.format('89201234567') // '+7 (920) 123-45-67'
```

```typescript
// hooks/useForm.ts
interface UseFormConfig<T extends Record<string, any>> {
  initialValues: T
  validators: Record<keyof T, (value: any) => string | null>
  onSubmit: (values: T) => Promise<void> | void
}

export function useForm<T extends Record<string, any>>({
  initialValues,
  validators,
  onSubmit
}: UseFormConfig<T>) {
  const [values, setValues] = useState<T>(initialValues)
  const [errors, setErrors] = useState<Record<keyof T, string | null>>({})
  const [isSubmitting, setIsSubmitting] = useState(false)

  const handleChange = useCallback(<K extends keyof T>(key: K, value: T[K]) => {
    setValues(prev => ({ ...prev, [key]: value }))
    // Автоматическая валидация onChange
    const error = validators[key]?.(value) ?? null
    setErrors(prev => ({ ...prev, [key]: error }))
  }, [validators])

  const handleSubmit = useCallback(async (e?: FormEvent) => {
    e?.preventDefault()

    // Валидация всех полей
    const newErrors = {} as Record<keyof T, string | null>
    Object.keys(validators).forEach(key => {
      const error = validators[key as keyof T]?.(values[key as keyof T])
      if (error) newErrors[key as keyof T] = error
    })
    setErrors(newErrors)

    if (Object.values(newErrors).every(err => err === null)) {
      setIsSubmitting(true)
      try {
        await onSubmit(values)
      } finally {
        setIsSubmitting(false)
      }
    }
  }, [values, validators, onSubmit])

  return {
    values,
    errors,
    isSubmitting,
    handleChange,
    handleSubmit,
    setValues,
    reset: () => setValues(initialValues)
  }
}

// ContactForm.tsx (73 строки вместо 207)
const ContactForm = ({ type, onSubmitSuccess }: ContactFormProps) => {

    initialValues: { name: '', phone: '', service: 'default' },

      name: (v) => v.trim().length >= 2 ? null : 'Минимум 2 символа',
```

```tsx
// components/ui/Button/Button.tsx
interface ButtonProps extends React.ButtonHTMLAttributes<HTMLButtonElement> {
  variant?: 'primary' | 'secondary' | 'danger' | 'ghost'
  size?: 'sm' | 'md' | 'lg'
  isLoading?: boolean
  leftIcon?: React.ReactNode
  rightIcon?: React.ReactNode
  fullWidth?: boolean
}

export const Button: React.FC<ButtonProps> = ({
  children,
  variant = 'primary',
  size = 'md',
  isLoading = false,
  leftIcon,
  rightIcon,
  fullWidth = false,
  className = '',
  disabled,
  ...props
}) => {
  const baseStyles = 'inline-flex items-center justify-center font-bold rounded transition-all active:scale-95 disabled:opacity-50 disabled:cursor-not-allowed min-h-[44px]'

  const variants = {
    primary: 'bg-[#1a224f] text-white hover:bg-[#2a3575]',
    secondary: 'bg-white text-[#1a224f] hover:bg-gray-100',
    danger: 'bg-[#d71e1e] text-white hover:bg-[#b01818]',
    ghost: 'bg-transparent text-[#1a224f] hover:bg-gray-100'
  }

  const sizes = {
    sm: 'px-3 py-2 text-sm',
    md: 'px-6 py-4 text-base',
    lg: 'px-10 py-5 text-lg'
  }

  return (
    <button
      className={`${baseStyles} ${variants[variant]} ${sizes[size]} ${fullWidth ? 'w-full' : ''} ${className}`}
      disabled={disabled || isLoading}
      {...props}

      {isLoading ? (
        <Spinner className="mr-2" />
      ) : leftIcon ? (
        <span className="mr-2">{leftIcon}</span>
      ) : null}

      {children}

      {rightIcon && <span className="ml-2">{rightIcon}</span>}
    </button>
  )
}
```

### **Пример 5: Context для модальных окон (решение prop drilling)**

```tsx
// contexts/ModalContext.tsx
interface ModalState {
  isOpen: boolean
  type: 'consultation' | 'engineer' | 'cost' | null
}

interface ModalContextType {
  modal: ModalState
  openModal: (type: ModalState['type']) => void
  closeModal: () => void
  successModal: SuccessModalState
  openSuccessModal: (props?: Partial<SuccessModalProps>) => void
  closeSuccessModal: () => void
}

const ModalContext = createContext<ModalContextType | null>(null)

export const ModalProvider: React.FC<{ children: ReactNode }> = ({ children }) => {
  const [modal, setModal] = useState<ModalState>({ isOpen: false, type: null })
  const [successModal, setSuccessModal] = useState<SuccessModalState>({
    isOpen: false,
    props: { title: 'Спасибо!', message: 'Ваша заявка принята.' }
  })

  const openModal = useCallback((type: ModalState['type']) => {
    setModal({ isOpen: true, type })
  }, [])

  const openSuccessModal = useCallback((props?: Partial<SuccessModalProps>) => {
    setSuccessModal(prev => ({
      ...prev,
      isOpen: true,
      props: { ...prev.props, ...props }
    }))
  }, [])

  return (
    <ModalContext.Provider value={{
      modal, openModal, closeModal: () => setModal(s => ({ ...s, isOpen: false })),
      successModal, openSuccessModal, closeSuccessModal: () => setSuccessModal(s => ({ ...s, isOpen: false }))
    }}>
      {children}
      <ModalRenderer /> {/* Рендерит модалки глобально */}
    </ModalContext.Provider>
  )
}
```
